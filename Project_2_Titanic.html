def group_age(df):
    age_cat=[]
    col_age=df.Age
    for x in col_age:
        if(x<1):
            age_cat.append('infant')
        elif(x<13):
            age_cat.append('child')
        elif(x<19):
            age_cat.append('teenager')
        elif(x<56):
            age_cat.append('adult')
        else:
            age_cat.append('the aged')
    return df.assign(Age_Cat = age_cat)

age_df = group_age(age_df)

def nor_group_fare(df, fare_col_name):
    fare = df[fare_col_name]
    fare_n = (fare-fare.mean())/fare.std()
    df = df.assign(Fare_N = fare_n)
    acf_cat=[]
    col_fare_n=df.Fare_N
    for x in col_fare_n:
        if(x<1):
            acf_cat.append('0_1')
        elif(x<2):
            acf_cat.append('1_2')
        elif(x<3):
            acf_cat.append('2_3')
        elif(x<4):
            acf_cat.append('3_4')
        else:
            acf_cat.append('>4')
    return df.assign(Fare_Cat = acf_cat)

acf_cat_df=nor_group_fare(acf_df,'Fare')
nor_fare_df = nor_group_fare(fare_df,'Fare')

def count_passenger(df, key):
    key_list = df[key].unique()
    new_df = pd.DataFrame({key: key_list})
    col_indexer = 'Passenger Count'
    for x in df[key]:
        row_indexer = new_df.index[new_df[key]==x]
        new_df.loc[row_indexer,col_indexer]=len(df[df[key] == x])
    return new_df

def create_df(df, key):
    key_list = df[key].unique()
    new_df = pd.DataFrame({'key': key_list})
    #df_pct = group_data(df, key)
    col_indexer = 'Percent'
    for x in key_list:
        row_indexer = new_df.index[new_df['key']==x]
        #new_df.loc[row_indexer,col_indexer]=df_pct.get(x)
        new_df.loc[row_indexer,col_indexer]=len(df[(df[key]==x) & (df['Survived']==1)])/len(df[df[key]==x])
    return new_df
